    MeshUVRemapper

    调用 UVRemapTrainner 中的方法优化纹理

    using System.Collections;
using System.Collections.Generic;
using UnityEngine;

public class MeshUVRemapper: MonoBehaviour
{
    public Mesh InputMesh;
    public Texture2D MainTexture;
    public Mesh OutputMesh;

    public bool AutoRun;
    [Range(1, 50)]
    public int StudyRateScale = 1;
    [Range(1, 10)]
    public int StudyPerFrame = 1;
    public float Loss = 0.0f;
    public float MinLoss = float.MaxValue;
    public int Step = 0;
    public int MinStep = 0;
    public Material SaveTextureMat;

    // 关键是这个类
    UVRemapTrainner trainner;
    void Start()
    {
        InputMesh = Instantiate(InputMesh);
        var uvs = InputMesh.uv;
        var vbo = new List<Vector3>();
        foreach(var uv in uvs)
        {
            vbo.Add(new Vector3(uv.x * 2 - 1, uv.y * 2 - 1, 0));
        }
        InputMesh.vertices = vbo.ToArray();

        trainner = new UVRemapTrainner(InputMesh, MainTexture);
        OutputMesh = Instantiate(InputMesh);
        OutputMesh.uv2 = OutputMesh.uv;
        OutputMesh.uv3 = trainner.CurrentDUDV;
        OutputMesh.colors = trainner.CurrentIMT;
        Loss = trainner.CurrentLoss;
        MinLoss = Loss;
    }

    [ContextMenu("StepOnce")]
    public void StepOnce()
    {
        trainner.StudyRateScale = StudyRateScale;
        for (int i = 0; i < StudyPerFrame; i++)
        {
            Step++;
            trainner.StepOnce();
            Loss = trainner.CurrentLoss;
            if (MinLoss > Loss)
            {
                MinLoss = Loss;
                MinStep = Step;
            }
        }
        OutputMesh.uv2 = trainner.CurrentUV;
        OutputMesh.uv3 = trainner.CurrentDUDV;
        OutputMesh.colors = trainner.CurrentIMT;
    }
    void Update()
    {
        if (Input.GetKeyUp(KeyCode.Space) || Input.GetKey(KeyCode.KeypadPlus) || AutoRun)
        {
            StepOnce();
        }
    }
}

    using System.Collections;

using System.IO;
using UnityEngine;

public class ReadMeshUV0 : MonoBehaviour
{
    public string UVTextPath;
    public Mesh RawMesh;
    [ContextMenu("Do")]
    void Do()
    {
        var uvs = File.ReadAllLines(UVTextPath);
        var mesh2 = Instantiate(RawMesh);
        if(uvs.Length == mesh2.vertexCount)
        {
            var uvd = new Vector2[uvs.Length];
            int i = 0;
            foreach(var uv in uvs)
            {
                var str = uv.Split(',');
                var u = float.Parse(str[0]);
                var v = 1.0f - float.Parse(str[1]);
                uvd[i] = new Vector2(u, v);
                i++;
            }
            mesh2.uv = uvd;
#if UNITY_EDITOR
            UnityEditor.AssetDatabase.CreateAsset(mesh2, "Assets/3DUVRemapper/newMesh.asset");
#endif
        }
    }
}

    private void Start()

        UVCamera.targetTexture = UVTexture;

        mainRT.name = "MainRT";
        DeltaCamera.targetTexture = DeltaRT;
        MainCamera.targetTexture = mainRT;
        DeltaMaterial.SetTexture("_Tex1", mainRT);
        DeltaMaterial.SetTexture("_Tex2", GroundTruthTexture);
        if(UseGPU)
        {
            mainTexRTSwap = new RenderTexture[2];
            backPropagateMaterial = new Material(Shader.Find("ConvexOpti/BackPropagate"));
        }
        else
        {
            cache_dtexelsArr = new SoftBilinearSampler.DSparseTexels[RenderSize * RenderSize];
        }
        cache_lossTexture = new Texture2D(RenderSize, RenderSize, TextureFormat.RGBAFloat, false, true);
    }
    private void OnDestroy()
    {
        dependenciesBuffer?.Release();
        offsetBuffer?.Release();
    }

    [ContextMenu("Init")]
    void Init()
    {
        PSNR = 0.0;
        IterIndex = 0;
        if(UseGPU)
        {
            if (mainTexRTSwap[0] == null || mainTexRTSwap[0].width != MainTexture.width || mainTexRTSwap[0].height != MainTexture.height) 
            {
                mainTexRTSwap[0] = new RenderTexture(MainTexture.width, MainTexture.height, 0, RenderTextureFormat.ARGB32, RenderTextureReadWrite.sRGB);
                mainTexRTSwap[0].name = nameof(mainTexRTSwap) + "0";
                mainTexRTSwap[1] = new RenderTexture(MainTexture.width, MainTexture.height, 0, RenderTextureFormat.ARGB32, RenderTextureReadWrite.sRGB);
                mainTexRTSwap[1].name = nameof(mainTexRTSwap) + "1";
            }
            mainTexRTSwapIndex = 0;
            Graphics.Blit(MainTexture, mainTexRTSwap[mainTexRTSwapIndex]);
            MainMaterial.mainTexture = mainTexRTSwap[mainTexRTSwapIndex];
            MainTexShowMaterial.mainTexture = mainTexRTSwap[mainTexRTSwapIndex];
            GenDependenciesBuffer();
        }
        else
        {
            if (AssetDatabase.IsMainAsset(MainTexture))
            {
                colors_MainTexture = MainTexture.GetPixels();
                MainTexture = new Texture2D(MainTexture.width, MainTexture.height, MainTexture.format, false, false);
                MainTextureSize = MainTexture.width;
                MainTexture.name = MainTexture.name + "(Clone)";
                MainTexture.filterMode = FilterMode.Bilinear;
                MainTexture.anisoLevel = 0;
                MainTexture.wrapMode = TextureWrapMode.Clamp;
                MainTexture.SetPixels(colors_MainTexture);
                MainTexture.Apply();
            }
            for (int i = 0; i < cache_dtexelsArr.Length; i++)
            {
                cache_dtexelsArr[i] = SoftBilinearSampler.DFDTexture(null, uvs_UVTexture[i], MainTextureSize, MainTextureSize);
            }
            MainMaterial.mainTexture = MainTexture;
            MainTexShowMaterial.mainTexture = MainTexture;
        }
        UpdateLoss();
    }

    private void GenDependenciesBuffer()
    {
        var dependenciesList = new List<DependenciesInfo>[MainTexture.width * MainTexture.height];
        for (int i = 0; i < dependenciesList.Length; i++)
        {
            dependenciesList[i] = new List<DependenciesInfo>();
        }
        for (uint i = 0; i < RenderSize * RenderSize; i++)
        {
            var texel = SoftBilinearSampler.DFDTexture(null, uvs_UVTexture[i], MainTextureSize, MainTextureSize);
            dependenciesList[texel.index1].Add(new DependenciesInfo() { index = i, dydx = texel.dydx1 });
            dependenciesList[texel.index2].Add(new DependenciesInfo() { index = i, dydx = texel.dydx2 });
            dependenciesList[texel.index3].Add(new DependenciesInfo() { index = i, dydx = texel.dydx3 });
            dependenciesList[texel.index4].Add(new DependenciesInfo() { index = i, dydx = texel.dydx4 });
        }
        var offsetList = new uint[dependenciesList.Length + 1];
        var dependenciesSingleList = new List<DependenciesInfo>();
        for (int i = 0; i < dependenciesList.Length; i++)
        {
            offsetList[i] = (uint)dependenciesSingleList.Count;
            dependenciesSingleList.AddRange(dependenciesList[i]);
        }
        offsetList[dependenciesList.Length] = (uint)dependenciesSingleList.Count;
        dependenciesBuffer?.Release();
        dependenciesBuffer = new ComputeBuffer(dependenciesSingleList.Count, Marshal.SizeOf(typeof(DependenciesInfo)));
        dependenciesBuffer.SetData(dependenciesSingleList.ToArray());
        offsetBuffer?.Release();
        offsetBuffer = new ComputeBuffer(offsetList.Length, sizeof(uint));
        offsetBuffer.SetData(offsetList);

        backPropagateMaterial.SetBuffer("_DependenciesBuffer", dependenciesBuffer);
        backPropagateMaterial.SetBuffer("_OffsetBuffer", offsetBuffer);
    }



    [ContextMenu("StepOnce")]
    void StepOnce()
    {
        if(UseGPU)
        {
            backPropagateMaterial.SetTexture("_DeltaTex", DeltaRT);
            backPropagateMaterial.SetTexture("_MainTex", mainTexRTSwap[mainTexRTSwapIndex]);
            backPropagateMaterial.SetFloat("_StudyRate", StudyRate);
            var nextRTIndex = (mainTexRTSwapIndex + 1) % 2;
            Graphics.Blit(mainTexRTSwap[mainTexRTSwapIndex], mainTexRTSwap[nextRTIndex], backPropagateMaterial);
            mainTexRTSwapIndex = nextRTIndex;
            MainMaterial.mainTexture = mainTexRTSwap[mainTexRTSwapIndex];
            MainTexShowMaterial.mainTexture = mainTexRTSwap[mainTexRTSwapIndex];
        }
        else
        {
            //Back Propagate
            for (int i = 0; i < cache_dtexelsArr.Length; i++)
            {
                var loss2 = colors_DeltaTexture[i];
                var dtexels = cache_dtexelsArr[i];
                colors_MainTexture[dtexels.index1] = BackPropagate(colors_MainTexture[dtexels.index1], dtexels.dydx1, loss2);
                colors_MainTexture[dtexels.index2] = BackPropagate(colors_MainTexture[dtexels.index2], dtexels.dydx2, loss2);
                colors_MainTexture[dtexels.index3] = BackPropagate(colors_MainTexture[dtexels.index3], dtexels.dydx3, loss2);
                colors_MainTexture[dtexels.index4] = BackPropagate(colors_MainTexture[dtexels.index4], dtexels.dydx4, loss2);
            }

            //Update Main Texture
            MainTexture.SetPixels(colors_MainTexture);
            MainTexture.Apply();
        }
        if (LogToPNG)
        {
            var oldRT = RenderTexture.active;
            RenderTexture.active = MainCamera.targetTexture;
            var texture2 = new Texture2D(RenderSize, RenderSize, TextureFormat.RGB24, false, true);
            texture2.ReadPixels(new Rect(0, 0, RenderSize, RenderSize), 0, 0);
            texture2.Apply();
            var bytes = texture2.EncodeToPNG();
            File.WriteAllBytes("CapturePNG/" + IterIndex + ".png", bytes);
            Destroy(texture2);
            RenderTexture.active = oldRT;

            bytes = MainTexture.EncodeToPNG();
            File.WriteAllBytes("CapturePNG/" + IterIndex + "_Main.png", bytes);

        }

        //Get Loss
        UpdateLoss();
        IterIndex++;
    }
    void UpdateLoss()
    {
        MainCamera.Render();
        DeltaCamera.Render();
        //if (UseGPU)

        }

            var index3 = cacheIndices[3 * i + 2];
            var area = HeronFormula(cacheVertices[index1], cacheVertices[index2], cacheVertices[index3]);

    private static float HeronFormula(Vector3 p1, Vector3 p2, Vector3 p3)
    {
        var a = (p1 - p2).magnitude;
        var b = (p2 - p3).magnitude;
        var c = (p3 - p1).magnitude;
        var s = 0.5f * (a + b + c);
        var area = Mathf.Sqrt(s * (s - a) * (s - b) * (s - c));
        return area;
    }

        var d3to2T = new DifferentiableMatrixNM(9, 4);
        d3to2T.Zero();
        d3to2T[0, 0] = 1;
        d3to2T[3, 1] = 1;
        d3to2T[1, 2] = 1;
        d3to2T[4, 3] = 1;
        return d3to2T;
    }
    private static double HeronFormula(DifferentiableMatrix3 points)
    {
        var p1 = points.Row0;
        var p2 = points.Row1;
        var p3 = points.Row2;
        var a = (p1 - p2).Magnitude;
        var b = (p2 - p3).Magnitude;
        var c = (p3 - p1).Magnitude;
        var s = 0.5 * (a + b + c);
        var area = Math.Sqrt(s * (s - a) * (s - b) * (s - c));
        return area;
    }
    public override DifferentiableScalar F(
        in DifferentiableMatrix3 x, in DifferentiableMatrix3 y,
        in DifferentiableMatrix3 z, in DifferentiableMatrix2 w)
    {
        var inv = DifferentiableFunction.Matrix3Inverse.F(x);
        var solution = DifferentiableFunction.Matrix3Multiply.F(inv, y);
        var solution2 = new DifferentiableMatrix2(solution[0], solution[3], solution[1], solution[4]);
        var e = affineToTr.F(solution2, w);
        var area = HeronFormula(z);
        return e * area;
    }

using System.Collections.Generic;
using System.Linq;
using UnityEngine;

public class UVRemapTrainner
{
    public float StudyRateScale { get; set; } = 1.0f;
    public int Steps { get; private set; }
    public Vector2[] CurrentUV { get; private set; }
    public Vector2[] CurrentDUDV { get; private set; }
    public Color[] CurrentIMT { get; private set; }
    public float CurrentLoss { get; private set; }

    private float StudyRate = 0.000001f;
    private Vector3[] cacheImts;
    private Vector3[] cacheVertices;
    private Vector2[] cacheUVs;
    private bool[] cacheEdgeU;
    private bool[] cacheEdgeV;
    private int[] cacheIndices;

    public UVRemapTrainner(Mesh inputMesh, Texture2D inputTexture)
    {
        this.cacheVertices = inputMesh.vertices;
        this.cacheIndices = inputMesh.triangles;
        this.cacheUVs = inputMesh.uv;
        this.CurrentUV = inputMesh.uv;
        this.cacheEdgeU = new bool[inputMesh.uv.Length];
        this.cacheEdgeV = new bool[inputMesh.uv.Length];
        for(int i = 0; i < inputMesh.uv.Length; i++)
        {
            var uv = this.cacheUVs[i];
            if(uv.x == 0.0f || uv.x == 1.0f) this.cacheEdgeU[i] = true;
            if(uv.y == 0.0f || uv.y == 1.0f) this.cacheEdgeV[i] = true;
        }
        GetAdjointTriangles();
        this.cacheImts = UVAtlasTool.ComputeIMTFromTexture(inputMesh, inputTexture);
        GetDUDV();

    private void GetAdjointTriangles()
    {
        var adjointTrigles = new List<Vector3Int>[cacheVertices.Length];
        for (int i = 0; i < cacheVertices.Length; i++)
        {
            adjointTrigles[i] = new List<Vector3Int>();
            var set = new HashSet<Vector3Int>();
            for (int j = 0; j < cacheIndices.Length / 3; j++)
            {
                var i1 = cacheIndices[3 * j + 0];
                var i2 = cacheIndices[3 * j + 1];
                var i3 = cacheIndices[3 * j + 2];
                if (i1 == i || i2 == i || i3 == i)
                {
                    set.Add(new Vector3Int(i1, i2, i3));
                }
            }
            adjointTrigles[i].AddRange(set.ToList());
        }
        cacheAdjiontTriangles = adjointTrigles;
    }

    public void StepOnce()
    {
        BackPropagate();
        GetDUDV();
        Steps++;
    }
    private void GetDUDV()
    {
        var currentUV = CurrentUV;
        var dudv = new Vector2[cacheVertices.Length];
        var imts = new Color[cacheVertices.Length];
        var count = new int[cacheVertices.Length];
        for (int i = 0; i < cacheVertices.Length; i++)
        {
            dudv[i] = new Vector2(0, 0);
            count[i] = 0;

        {
            var index1 = cacheIndices[3 * i + 0];
            var index2 = cacheIndices[3 * i + 1];
            var index3 = cacheIndices[3 * i + 2];

                PositionY = cacheVertices[index1].y,

                TexCoord2V = currentUV[index1].y,
                dEdU = 0,
                dEdV = 0,
            };
            var v2 = new UVRemapper.VertexData()
            {
                PositionX = cacheVertices[index2].x,
                PositionY = cacheVertices[index2].y,
                PositionZ = cacheVertices[index2].z,

            {
                PositionX = cacheVertices[index3].x,
                PositionY = cacheVertices[index3].y,
                PositionZ = cacheVertices[index3].z,
                TexCoordU = cacheUVs[index3].x,
                TexCoordV = cacheUVs[index3].y,
                TexCoord2U = currentUV[index3].x,
                TexCoord2V = currentUV[index3].y,

                b = cacheImts[i].y,
                c = cacheImts[i].z
            };
            try
            {
                var error = UVRemapper.GetSurfaceError(ref v1, ref v2, ref v3, imt);
                dudv[index1] += new Vector2((float)v1.dEdU, (float)v1.dEdV);
                dudv[index2] += new Vector2((float)v2.dEdU, (float)v2.dEdV);
                dudv[index3] += new Vector2((float)v3.dEdU, (float)v3.dEdV);
                totalError += (float)error;
                imts[index1] += new Color(cacheImts[i].x, cacheImts[i].y, cacheImts[i].z, (float)error);
                imts[index2] += new Color(cacheImts[i].x, cacheImts[i].y, cacheImts[i].z, (float)error);
                imts[index3] += new Color(cacheImts[i].x, cacheImts[i].y, cacheImts[i].z, (float)error);

    float UVArea(Vector2 p1, Vector2 p2, Vector2 p3, bool cw = true)
    {
        var a = p2 - p1;
        var b = p3 - p2;
        return (cw ? 0.5f : -0.5f) * (a.x * b.y - a.y * b.x);
    }
    private void BackPropagate()
    {
        var dudv = CurrentDUDV;
        var uv = CurrentUV;
        var rndIndex = RandomSort(dudv.Length);
        for (int i = 0; i < dudv.Length; i++)
        {
            var index = rndIndex[i];
            var uvNew = uv[index] - StudyRate * StudyRateScale * dudv[index];
            uvNew.x = Mathf.Clamp01(uvNew.x);
            uvNew.y = Mathf.Clamp01(uvNew.y);
            if (cacheEdgeU[index]) uvNew.x = uv[index].x;
            if (cacheEdgeV[index]) uvNew.y = uv[index].y;

            var adjionts = cacheAdjiontTriangles[index];
            bool error = false;
            for (int j = 0; j < adjionts.Count; j++)
            {
                var adj = adjionts[j];
                var uv1 = adj.x == index ? uvNew : uv[adj.x];
                var uv2 = adj.y == index ? uvNew : uv[adj.y];
                var uv3 = adj.z == index ? uvNew : uv[adj.z];
                var area = UVArea(uv1, uv2, uv3, false);
                if(area <= 0.0f)
                {
                    error = true;
                    Debug.LogError("Area Error");
                    break;
                }
            }
            if(!error)
            {
                uv[index] = uvNew;
            }
        }
        CurrentUV = uv;
    }
    private static int[] RandomSort(int length)
    {
        var indices = new int[length];
        for(int i = 0; i < length; i++)
        {
            indices[i] = i;
        }
        for (int i = 0; i < length - 1; i++)
        {
            var rnd = UnityEngine.Random.Range(i, length);
            indices[i] = indices[rnd];
            indices[rnd] = i;
        }
        return indices;
    }
